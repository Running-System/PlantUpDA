\section{Sensor Technologies for Plant Monitoring (approx. 4 Pages)}

\subsection{Overview of Common Sensor Types}
\begin{itemize}
    \item \textbf{Moisture:} Resistive (corrodes) vs. Capacitive (durable).
    \item \textbf{Air:} Combined sensors (DHT22, BME280).
    \item \textbf{Light:} Photoresistors (cheap) vs. Digital (Lux-calibrated).
\end{itemize}

\subsection{Limitations and Typical Error Sources}
\begin{itemize}
    \item Soil usually not uniform (wet spots vs dry spots).
    \item Sensors drift over time.
    \item Electrical noise in readings.
\end{itemize}

\subsection{Selection Criteria}
\begin{itemize}
    \item \textbf{Decision:} Capacitive sensors for longevity.
    \item \textbf{Voltage:} 3.3V compatibility for ESP32.
    \item \textbf{Cost:} Good balance between price and accuracy.
\end{itemize}
